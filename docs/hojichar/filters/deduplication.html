 18    生成されたハッシュは `Document.dedup_lsh` 属性に文字列のリストとして保存されます。

 20    """
 21
 22    N_MINHASH = 200
 23    N_GRAM = 5
 24    N_BUCKETS = 20
 25    BUCKET_SIZE = 10
 26
 27    def __init__(self, *args, **kwargs) -> None:
 28        super().__init__(*args, **kwargs)

 30    @staticmethod
 31    def n_gram_tokenize(text: str, n: int) -> List[str]:
 32        """
 33        Tokenize a string into n-gram tokens.
 34
 35        >>> GenerateDedupLSH().n_gram_tokenize("おはようございます。", 3)
 36        ['おはよ', 'はよう', 'ようご', 'うござ', 'ござい', 'ざいま', 'います', 'ます。']
 37
 38        >>> GenerateDedupLSH().n_gram_tokenize("おはよ", 5)
 39        ['おはよ']
 40        """
 41        if len(text) < n:
 42            return [text]
 43        else:
 44            return [text[idx : idx + n] for idx in range(len(text) - n + 1)]

 46    @staticmethod
 47    def hashfunc_signed_32_from_seed(seed: int) -> Callable[[str], int]:
 48        return lambda text: mmh3.hash(text, seed, signed=True)

 50    @staticmethod
 51    def get_minhash(tokens: List[str], hashfunc: Callable[[str], int]) -> int:
 52        return min([hashfunc(text) for text in tokens])

 54    def calc_lsh(self, text: str) -> List[str]:
 55        """
 56        テキストから重複処理のためのハッシュ値のシーケンスを生成します.
 57        2文書間で, 少なくとも1つ同じハッシュ値を保つ場合, 文書は重複と判断します.
 58        (参考文献)
 59        https://arxiv.org/abs/2107.06499 , Appendix A
 60        http://infolab.stanford.edu/~ullman/mmds/ch3n.pdf Chapter 3.3
 61
 62        Args:
 63            text (str): 入力テキスト
 64
 65        Returns:
 66            List[str]: ハッシュ値のリスト, 各ハッシュ値は
 67            '0+07ad0b7b163f434643387f3f4799a2d466bccd0c' のような形式で,
 68            先頭2文字は何番目のハッシュ値かを表す.
 69            これにより, 重複処理ハッシュを一つのハッシュテーブルにプールすることで重複処理ができる.
 70        """
 71        # N_MINHASH 個の mmh3 ハッシュ値 から最終的に N_BUKET 個の重複処理ハッシュを計算する
 72        fingerprints = []
 73        for seed in range(self.N_MINHASH):
 74            hashfunc = self.hashfunc_signed_32_from_seed(seed)
 75            minhash = self.get_minhash(self.n_gram_tokenize(text, n=self.N_GRAM), hashfunc)
 76            fingerprints.append(minhash)
 77
 78        # 速度のためにリスト内包で書いており, 可読性低め
 79        # 各 fingerprint を 符号あり -> 符号なしにして, 16進数表記にして, 下四桁をバケットサイズ個ずつ連結している
 80        # HACK 符号なしにする操作は不要, 既存のハッシュ値との互換性を破壊して再実装する際は取り除いてよい
 81        # TODO Python だとオーバーヘッドが大きいので, C++ で実装しなおす
 82        lshs = []
 83        for bucket_idx in range(self.N_BUCKETS):
 84            lshs.append(
 85                str(bucket_idx)
 86                + "+"
 87                + "".join(
 88                    [
 89                        format(fingerprints[fp_idx] & 0xFFFFFFFF, "08x")[-4:]
 90                        for fp_idx in range(
 91                            bucket_idx * self.BUCKET_SIZE, (bucket_idx + 1) * self.BUCKET_SIZE
 92                        )
 93                    ]
 94                )
 95            )
 96
 97        return lshs

183class LSHDeduplicator(Filter):
184    """
185    `hojichar.filters.document_filter.GenerateDedupLSH` で生成したハッシュ値を基に重複判定をします。
186    対象コーパスが約 10^6 以下 (〜数十GBが目安) であれば、事前処理なしで重複処理が可能です。
187    事前処理なしで(オンラインに)重複除去を行う場合、`online_dedup` フラグを `True` にします。`online_dedup` のデフォルト値は `True` です。
188
189    より大きなコーパスではすべての文書のハッシュ値をメモリに載せることが困難になります。重複文書は文書全体の数パーセントだと考えられるため、重複文書の
190    ハッシュ値のみをブラックリストとしてファイルから読み込むことで数百GBのコーパスに対して処理ができます。
191    `blacklist_path` に指定したパスのファイルから重複ハッシュ値を読み込みます。ブラックリストのファイルは1行おきに
192    `GenerateDedupLSH`で生成したハッシュ値が記録されていることを仮定します。
193    `store_blacklist` フラグを `True` にすると、重複のあったハッシュ値を `LSHDeduplicator.blacklist` 属性に 文字列の集合として記録します。
194    このオプションは、ブラックリストのハッシュ値のファイルの作成時などに有効です。`store_blacklist` フラグのデフォルト値は `False`です。
195    """
196
197    def __init__(
198        self,
199        online_dedup=True,
200        blacklist_path="",
201        store_blacklist=False,
202        *args,
203        **kwargs,
204    ) -> None:
205        super().__init__(*args, **kwargs)
206        self.online_dedup = online_dedup
207        self.store_blacklist = store_blacklist
208        self.seen = set()
209        if blacklist_path:
210            with open(blacklist_path) as fp:
211                for line in fp:
212                    lsh = line.strip()
213                    self.seen.add(lsh)
214
215        if store_blacklist:
216            self.blacklist = copy.copy(self.seen)
217
218    def apply(self, doc: Document) -> Document:
219        """
220        >>> d1 = GenerateDedupLSH().apply(Document("Hello, World."))
221        >>> d2 = GenerateDedupLSH().apply(Document("吾輩は猫である。名前はまだ無い。どこで生まれたかとんと見当がつかぬ。"))
222        >>> d3 = GenerateDedupLSH().apply(Document("吾輩は鳥である。名前はまだ無い。どこで生まれたかとんと見当がつかぬ。"))
223        >>> deduplicator = LSHDeduplicator()
224        >>> deduplicator.apply(d1).is_rejected
225        False
226        >>> deduplicator.apply(d2).is_rejected
227        False
228        >>> deduplicator.apply(d3).is_rejected
229        True
230        """
231        lshs = doc.dedup_lsh
232        if len(lshs) == 0:
233            assert ValueError(
234                "LSHs for deduplication are not caluculated. Filter \
235                    `GenerateDedupLSH` must be composed before this filter."
236            )
237
238        for lsh in lshs:
239            if lsh in self.seen:
240                doc.is_rejected = True
241                if self.store_blacklist:
242                    self.blacklist.add(lsh)
243
244            if self.online_dedup:
245                self.seen.add(lsh)
246
247        return doc

55    def calc_lsh(self, text: str) -> List[str]:
56        """
57        テキストから重複処理のためのハッシュ値のシーケンスを生成します.
58        2文書間で, 少なくとも1つ同じハッシュ値を保つ場合, 文書は重複と判断します.
59        (参考文献)
60        https://arxiv.org/abs/2107.06499 , Appendix A
61        http://infolab.stanford.edu/~ullman/mmds/ch3n.pdf Chapter 3.3
62
63        Args:
64            text (str): 入力テキスト
65
66        Returns:
67            List[str]: ハッシュ値のリスト, 各ハッシュ値は
68            '0+07ad0b7b163f434643387f3f4799a2d466bccd0c' のような形式で,
69            先頭2文字は何番目のハッシュ値かを表す.
70            これにより, 重複処理ハッシュを一つのハッシュテーブルにプールすることで重複処理ができる.
71        """
72        # N_MINHASH 個の mmh3 ハッシュ値 から最終的に N_BUKET 個の重複処理ハッシュを計算する
73        fingerprints = []
74        for seed in range(self.N_MINHASH):
75            hashfunc = self.hashfunc_signed_32_from_seed(seed)
76            minhash = self.get_minhash(self.n_gram_tokenize(text, n=self.N_GRAM), hashfunc)
77            fingerprints.append(minhash)
78
79        # 速度のためにリスト内包で書いており, 可読性低め
80        # 各 fingerprint を 符号あり -> 符号なしにして, 16進数表記にして, 下四桁をバケットサイズ個ずつ連結している
81        # HACK 符号なしにする操作は不要, 既存のハッシュ値との互換性を破壊して再実装する際は取り除いてよい
82        # TODO Python だとオーバーヘッドが大きいので, C++ で実装しなおす
83        lshs = []
84        for bucket_idx in range(self.N_BUCKETS):
85            lshs.append(
86                str(bucket_idx)
87                + "+"
88                + "".join(
89                    [
90                        format(fingerprints[fp_idx] & 0xFFFFFFFF, "08x")[-4:]
91                        for fp_idx in range(
92                            bucket_idx * self.BUCKET_SIZE, (bucket_idx + 1) * self.BUCKET_SIZE
93                        )
94                    ]
95                )
96            )
97
98        return lshs

hojichar.filters.document_filter.GenerateDedupLSH で生成したハッシュ値を基に重複判定をします。 対象コーパスが約 10^6 以下 (〜数十GBが目安) であれば、事前処理なしで重複処理が可能です。 事前処理なしで(オンラインに)重複除去を行う場合、online_dedup フラグを True にします。online_dedup のデフォルト値は True です。
