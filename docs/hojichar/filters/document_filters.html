 99class JSONLoader(Filter):
100    """
101    テキストを Json として解釈し, `key` で指定した要素を文字列として
102    doument に格納します.デフォルトの `key` は 'text' です.
103
104    Json の読み込み, あるいは `key` の読み込みに失敗した際には例外を送出します.
105    これらを無視する場合は, `ignore=True` にします. その際, 読み込みに失敗
106    したドキュメントは破棄されます.
107    """
108
109    def __init__(self, key="text", ignore=False, *args, **kwargs):
110        super().__init__(*args, **kwargs)
111        self.key = key
112        self.ignore = ignore
113
114    def apply(self, document: Document) -> Document:
115        """
116        >>> JSONLoader()( '{"text": "hello, world", "words": 2}' )
117        'hello, world'
118
119        >>> JSONLoader()( '{"text": hello, world ....' ) # Broken JSON
120        Traceback (most recent call last):
121            ...
122        json.decoder.JSONDecodeError: Expecting value: line 1 column 10 (char 9)
123
124        >>> JSONLoader()( '{"words": 2}' )
125        Traceback (most recent call last):
126            ...
127        KeyError: 'text'
128
129        >>> JSONLoader(ignore=True).apply(Document('{"text": hello, world ....' )).is_rejected
130        True
131        """
132        try:
133            data = json.loads(document.text)
134            document.text = str(data[self.key])
135        except Exception as e:
136            if self.ignore:
137                document.is_rejected = True
138                return document
139            else:
140                raise e
141
142        return document

172class NgWordsFilterJa(Filter):
173    """
174    日本語のNGワード(および不適切語)を含む文書を破棄します.
175    `dict_path` で指定したファイルから, キーワードのリストを得ます.
176    ファイルは単語が改行で羅列されたテキストファイルです.
177
178    `ignore_confused` を `True` にすると,
179    偽陽性を軽減するために, カタカナのNGワードは前後にカタカナが無い場合のみNG判定されます.
180    デフォルト値は `False` です.
181    """
182
183    def __init__(self, dict_path, ignore_confused=False, *args, **kwargs):
184        super().__init__(*args, **kwargs)
185
186        with open(dict_path, encoding="utf-8") as fp:
187            ng_words = fp.read().split("\n")
188        ng_words = [w.strip() for w in ng_words if not len(w) == 0]
189
190        if ignore_confused:
191            words_katakana = []
192            words_not_katakana = []
193            for w in ng_words:
194                if re.fullmatch(r"[ァ-ヴー]+", w):
195                    words_katakana.append(re.escape(w))
196                else:
197                    words_not_katakana.append(re.escape(w))
198            katakana_pat = "|".join(words_katakana)
199            katakana_pat = rf"(?<![ァ-ヴー])({katakana_pat})(?![ァ-ヴー])"
200            pat = "|".join(words_not_katakana) + "|" + katakana_pat
201            self.keyword_pat = re.compile(pat)
202        else:
203            ng_words = [re.escape(w) for w in ng_words]
204            pat = "|".join(ng_words)
205            self.keyword_pat = re.compile(pat)
206
207    def apply(self, doc: Document) -> Document:
208        if self.keyword_pat.search(doc.text):
209            doc.is_rejected = True
210        return doc

463class HeaderFooterTagsRemover(Filter):
464    """
465    ドキュメントの冒頭・末尾のトークンを調査し, ヘッダー・フッダー的な
466    タグが存在していた場合, そのトークンを除去します.
467
468    このフィルタを通す前に, 事前にセンテンスレベルにトーカナイズしておいてください.
469    このフィルタでは Document.token にのみ変更が加えられるので, 出力前 あるいは 下流フィルタで
470    Document.text に変更を加える前にトークンをマージしておいてください.
471    """
472
473    def __init__(
474        self,
475        dict_path=BASE_PATH / "dict/header_footer_keywords_ja.txt",
476        *args,
477        **kwargs,
478    ) -> None:
479        super().__init__(*args, **kwargs)
480
481        with open(dict_path) as fp:
482            keywords = fp.read().split("\n")
483        keywords = [re.escape(w.strip()) for w in keywords if not len(w) == 0]
484        self.keyword_pat = re.compile(r"|".join(keywords))
485
486    def apply(self, doc: Document) -> Document:
487        if len(doc.tokens) == 0:
488            return doc
489
490        lookup_size = 0
491        if 1 <= len(doc.tokens) < 4:
492            lookup_size = 1
493        elif 4 <= len(doc.tokens) < 6:
494            lookup_size = 2
495        elif 6 <= len(doc.tokens):
496            lookup_size = 3
497
498        for i in range(lookup_size):
499            if self.should_drop_token(doc.tokens[i]):
500                doc.tokens[i].is_rejected = True
501            if self.should_drop_token(doc.tokens[-(i + 1)]):
502                doc.tokens[i].is_rejected = True
503
504        return doc
505
506    def should_drop_token(self, token: Token) -> bool:
507        """
508        >>> HeaderFooterTagsRemover().should_drop_token(Token("<TEST_STRING_OF_KEYWORD>"))
509        True
510
511        >>> HeaderFooterTagsRemover().should_drop_token(Token("ほうじ茶"))
512        False
513
514        Comment.
515        Original legacy code removed a pattern r"« _ | Main | _ »" .
516        In the pattern, "|" is not escaped, so **ANY** string was eliminated.
517        It seems unintended behavior, so I fix this.
518        """
519        if self.keyword_pat.match(token.text):
520            return True
521        else:
522            return False
